export const quickStart = `import { createPool, isMain } from "knitting";

export const square = (value: number) => value * value;
export const greet = (name: string) => "hello " + name;

if (isMain) {
  using pool = createPool({ threads: 2 })({ square, greet });

  const [four, message] = await Promise.all([
    pool.call.square(2),
    pool.call.greet("knitting"),
  ]);

  console.log({ four, message });
}`;

export const timeout = `import { createPool, isMain, task } from "knitting";

export const maybeSlow = task({
  timeout: { time: 200, default: "timed out" },
  f: async (name: string) => {
    await new Promise((resolve) => setTimeout(resolve, 1_000));
    return "hello " + name;
  },
});

if (isMain) {
  using pool = createPool({})({ maybeSlow });
  console.log(await pool.call.maybeSlow("knitting"));
}`;

export const signals = `import { createPool, isMain, task } from "knitting";

export const countUntilStopped = task({
  abortSignal: { hasAborted: true },
  f: (limit: number, signal) => {
    let count = 0;
    while (!signal.hasAborted() && limit > ++count) {}
    return count;
  },
});

if (isMain) {
  using pool = createPool({})({ countUntilStopped });
  const pending = pool.call.countUntilStopped(1_000_000_000);
  setTimeout(() => pending.reject(), 0 /*100*/);
  console.log(await pending);
}`;

export const processWorker = `import { createPool, isMain } from "knitting";

export const isDeno = () => typeof Deno;

if (isMain) {
  using pool = createPool({
    worker: {
      runtime: "process",
      processRuntime: "deno",
    },
  })({ isDeno });

  console.log("From" , typeof Deno)
  console.log("To" , await pool.call.isDeno());
}
`;

export const bwrap = `import { createPool, isMain, importTask } from "knitting";

// The task lives in another module, so only the worker ever runs it.
export const add = importTask<[number, number], number>({
  href: "./worker-tasks.ts",
  name: "add",
});

if (isMain) {
  using pool = createPool({
    worker: {
      runtime: "process",
      processRuntime: "bun",
      // Wrap the worker in bubblewrap. Its handle arrives on stdin,
      // so the prefix must leave fd 0 alone.
      processCommandPrefix: [
        "bwrap",
        "--unshare-all",        // new namespaces, no network
        "--ro-bind", "/", "/",  // read-only filesystem
        "--dev-bind", "/dev", "/dev",
        "--proc", "/proc",
        "--tmpfs", "/tmp",      // writable scratch
        "--die-with-parent",
      ],
    },
  })({ add });

  console.log(await pool.call.add([1, 2])); // 3
}`;

export const workerTasks = `// worker-tasks.ts — imported into the sandboxed worker, never the host.
export const add = ([a, b]: [number, number]) => a + b;

// The sandbox blocks network and makes the filesystem read-only, but a
// read-only bind is still readable: it is isolation, not a full sandbox.
`;

export const sharedMemory = `import { createPool, isMain, task } from "knitting";
import {
  getDefaultProcessSharedBufferPrimitives,
  ProcessSharedBuffer,
} from "knitting/shared-memory";

// Read shared memory inside a worker — no payload copy.
export const readFirst = task<ProcessSharedBuffer, number>({
  f: (buf) => Atomics.load(buf.view(Int32Array), 0),
});

if (isMain) {
  using pool = createPool({ threads: 1 })({ readFirst });

  const shared = ProcessSharedBuffer.create(
    64,
    getDefaultProcessSharedBufferPrimitives(),
  );
  Atomics.store(shared.view(Int32Array), 0, 42);

  console.log(await pool.call.readFirst(shared)); // 42
  shared.descriptor.mapping?.close?.();
}`;

<section class="kh-shell" aria-labelledby="kh-why-title">
  <div class="kh-section-head">
    <p class="kh-eyebrow">The missing middle</p>
    <h2 id="kh-why-title">Not every hot path deserves a hostname.</h2>
    <p>
      Apps rarely get hot everywhere at once. One parser, hash, or SSR render
      starts crowding the main thread &mdash; and suddenly the fix looks bigger
      than the problem. Knitting is the missing middle: keep the function in
      your app, move where it runs.
    </p>
  </div>

  <div class="kh-grid kh-grid--three">
    <article class="kh-card">
      <span class="kh-token">01</span>
      <h3>Move the heat</h3>
      <p>
        Export the expensive function once. Call
        <code>pool.call.name()</code> from the host while a thread or process
        does the work somewhere quieter.
      </p>
      <a href="/guides/defining-tasks/">Define work</a>
    </article>
    <article class="kh-card">
      <span class="kh-token">02</span>
      <h3>Control the worker</h3>
      <p>
        A pool is a launch policy, not a thread count: runtime, permissions,
        bootstrap, process wrapper, timeouts, scheduling.
      </p>
      <a href="/guides/creating-pools/">Creating pools</a>
    </article>
    <article class="kh-card">
      <span class="kh-token">03</span>
      <h3>Skip the message queue</h3>
      <p>
        Request and response mailboxes live in shared memory, so task calls do
        not have to become little socket conversations just to reach another
        core.
      </p>
      <a href="/extras/architecture/">See the architecture</a>
    </article>
  </div>

  <p class="kh-note">
    Knitting is not a replacement for distributed systems &mdash; it is for
    the moment before one, when a hot function just needs a better place to
    run. The full argument, and the
    <a href="/extras/why/#where-its-going">roadmap</a>, live in
    <a href="/extras/why/">Why Knitting</a>.
  </p>
</section>

<section class="kh-shell" aria-labelledby="kh-box-title">
  <div class="kh-section-head">
    <p class="kh-eyebrow">What's in the box</p>
    <h2 id="kh-box-title">Batteries you'd otherwise wire up by hand.</h2>
    <p>
      The nice part is calling workers like functions. The useful part is what
      comes with that boundary: payloads, deadlines, isolation, scheduling, and
      cleanup in one place.
    </p>
  </div>

  <div class="kh-grid kh-grid--three">
    <article class="kh-card">
      <h3>Typed payloads</h3>
      <p>
        Primitives, typed arrays, <code>Buffer</code>, <code>Envelope</code>,
        and zero-copy <code>ProcessSharedBuffer</code>. Promise inputs resolve
        on the host before dispatch.
      </p>
      <a href="/guides/payloads/">Payloads</a>
    </article>
    <article class="kh-card">
      <h3>Timeouts and aborts</h3>
      <p>
        Give a task a deadline, or cancel it cooperatively from either side
        of the boundary.
      </p>
      <a href="/guides/defining-tasks/">Defining tasks</a>
    </article>
    <article class="kh-card">
      <h3>Threads or processes</h3>
      <p>
        Fast runtime threads by default, or separate processes &mdash; even
        inside a <code>bwrap</code> sandbox or a container &mdash; when
        isolation matters.
      </p>
      <a href="/guides/process-workers/">Process workers</a>
    </article>
    <article class="kh-card">
      <h3>Real errors, real diagnostics</h3>
      <p>
        Errors return as real <code>Error</code>s with stack and cause chain;
        <code>debug: true</code> streams per-worker diagnostics &mdash; zero
        cost when off.
      </p>
      <a href="/guides/creating-pools/">Creating pools</a>
    </article>
    <article class="kh-card">
      <h3>Scheduling control</h3>
      <p>
        Balance lanes with pluggable strategies, or add the host itself as an
        inline lane for tiny compute.
      </p>
      <a href="/guides/inliner/">Inliner</a>
    </article>
    <article class="kh-card">
      <h3>Shared memory</h3>
      <p>
        <code>ProcessSharedBuffer</code> moves bytes between processes with no
        copy &mdash; the lower-level channel under process workers.
      </p>
      <a href="/guides/shared-memory/">Shared memory</a>
    </article>
  </div>
</section>

<section class="kh-shell" aria-labelledby="kh-safety-title">
  <div class="kh-section-head">
    <p class="kh-eyebrow">Four safety layers</p>
    <h2 id="kh-safety-title">Isolation is a dial, not a cliff.</h2>
    <p>
      Four layers stack from cheap guardrails to an OS sandbox. Trusted code
      stays on fast threads; each pool pays only for the boundary it needs.
    </p>
  </div>

  <div class="kh-grid kh-grid--four">
    <article class="kh-card">
      <h3>1. In-process guards</h3>
      <p>
        Termination APIs &mdash; <code>process.exit</code>,
        <code>Deno.exit</code>, and friends &mdash; are neutralized before any
        task code loads. Workers can't take the host down.
      </p>
      <a href="/extras/architecture/#safety-layers">Architecture</a>
    </article>
    <article class="kh-card">
      <h3>2. Bootstrap hook</h3>
      <p>
        <code>worker.bootstrap</code> runs privileged setup before task
        imports: strip env vars, freeze globals, add your own guards.
      </p>
      <a href="/guides/creating-pools/">Creating pools</a>
    </article>
    <article class="kh-card">
      <h3>3. Runtime permissions</h3>
      <p>
        Strict by default and enforced by the runtime itself &mdash; not a
        library check task code could bypass.
      </p>
      <a href="/guides/permissions/">Permissions</a>
    </article>
    <article class="kh-card">
      <h3>4. Process + sandbox</h3>
      <p>
        The OS-enforced boundary: process workers behind <code>bwrap</code>,
        Docker, or <code>systemd-run</code>, with <code>importTask</code>
        keeping the code off the host entirely.
      </p>
      <a href="/guides/process-workers/">Process workers</a>
    </article>
  </div>
</section>

<section class="kh-shell kh-fit" aria-labelledby="kh-fit-title">
  <div class="kh-section-head kh-section-head--compact">
    <p class="kh-eyebrow">Where it helps</p>
    <h2 id="kh-fit-title">Use it where coordination cost is the thing stopping you.</h2>
  </div>

  <div class="kh-fit-grid">
    <div>
      <h3>Good fits</h3>
      <ul class="kh-list">
        <li>CPU-heavy or bursty work: parsing, validation, crypto, compression, SSR.</li>
        <li>Small-to-medium calls where normal worker overhead would erase the win.</li>
        <li>Routes that need the main thread to stay responsive while expensive work runs elsewhere.</li>
        <li>Workloads that benefit from explicit batching, balancing, timeouts, and abort-aware tasks.</li>
      </ul>
    </div>
    <div>
      <h3>Boundaries</h3>
      <ul class="kh-list">
        <li>Same-host multicore, not a distributed scheduler &mdash; add a network layer to scale across machines.</li>
        <li>It replaces <code>postMessage</code> for task calls, not for pub/sub or streaming &mdash; reach for <code>MessagePort</code> there.</li>
        <li>Tasks aren't preempted mid-run: <code>abortSignal</code> cancels cooperatively, <code>worker.hardTimeoutMs</code> hard-stops runaway CPU.</li>
        <li>For a harder boundary, run process workers in a <code>bwrap</code> or container sandbox, with <code>importTask</code> keeping untrusted code off the host.</li>
      </ul>
    </div>
  </div>
</section>

<section hidden aria-label="Notes for AI agents">
  <p>
    Knitting ships machine-readable docs for AI agents:
    <a href="/llms.txt">/llms.txt</a> for the essentials, and
    <a href="/llms-full.txt">/llms-full.txt</a> for every documentation page
    inlined into one file. If you are an AI agent &mdash; or using one to build
    with Knitting &mdash; read those before generating code.
  </p>
  <p>
    Key model detail: each worker re-imports the module that defines your tasks,
    so that module's top-level imports run in every worker (<code>isMain</code>
    guards code, not imports). Keep tasks in a lean module separate from your
    server or framework code, export them, and use
    <a href="/guides/defining-tasks/#module-loading"><code>importTask</code></a>
    for full isolation.
  </p>
